```js title="What is our action?"

    // Recursion

sum([1], 0) // 0

```

```js title="What's the purpose of the function? Addition!"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return ? + ?

    // Recursion
}

sum([2,3,4,5], 3) // 9

```

```js title="Find the element we're on and add it to something"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[?] + ?

    // Recursion
}

sum([2,3,4,5], 3) // 9

```

```js title="Use n to find the index of the current element"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[n - 1] + ?

    // Recursion
}

sum([2,3,4,5], 3) // 9

```

```js title="Why do we have to subtract 1 from n?"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[n - 1] + ?

    // Recursion
}

sum([2,3,4,5], 3) // 9

```

```js title="Blasted arrays are zero base indexed!"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[n - 1] + ?

    // Recursion
}

sum([2,3,4,5], 3) // 9

```

```diff title="If want the third element, you really want the element with the index of 2"
```

```diff 11[10,10] title="When n is 3, 4 is our first element. It's index is n - 1 = 2"
```

```js title="Add the element we're on, to the sum of the elements before it!"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[n - 1] + sum(arr, ?)

    // Recursion
}

sum([2,3,4,5], 3) // 9

```
```diff title="What are we using to move through the array? Our parameter n!"
```

```js title="This is how we move backward through the array"

function sum(arr, n) {
    if (n === 0)
        return 0
    
    return arr[n - 1] + sum(arr, n - 1)
}

sum([2,3,4,5], 3) // 9

```

```diff title="As n is decremented, so does our index in the array"
```
